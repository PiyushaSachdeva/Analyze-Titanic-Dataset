### Function to calculate chi square statistics for multiple independent variables
def calculate_chi_square_stats(independent_variable, alpha, degrees_of_freedom ):
    #Overall_rate
    survived= titanic_df['Survived'].mean()
    #Observed
    observed=titanic_df.groupby([independent_variable]).sum()['Survived']
    #expected
    expected=titanic_df.groupby([independent_variable]).count()['Survived']*survived
    #Chi square statistics
    chiStats= scipy.stats.chisquare(observed,expected,ddof=0)
    return chiStats

#Function to plot bar graph 
%pylab inline

def bar_graph_creation(groupby_var1, groupby_var2, agg_var, agg_type):
    survivability_count_per_var=titanic_df.groupby([groupby_var1,groupby_var2]).agg({agg_var:agg_type})
    survivability_percent_per_var=survivability_count_per_var.groupby(level=0).apply(lambda x:100 * x / float(x.sum()))
    survivability_reset_index=survivability_percent_per_var.reset_index()
    plt.ylim(0,100)
    return sns.barplot(x=groupby_var1, y=agg_var, hue='Survived',data=survivability_reset_index)
